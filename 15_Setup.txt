/**
 * 15_Setup.gs
 * Productivity OS v4.7 — 一键初始化
 *
 * 【V4.7 新增】setupSheets()/repairSheetHeaders() 的 Tasks/ActiveTasks/
 * ArchiveTasks 三张表表头新增 due_time（'HH:mm' 或 ''）、due_datetime
 * （'yyyy-MM-ddTHH:mm:ss' 或 ''，派生字段）两列，紧跟在 due_date 之后
 * ——Due Time Support，完整设计见 00_Architecture_Review.gs「七、
 * Review #3」，Carson 2026-07-13 批准。已有部署走 11_ProjectionRebuilder.gs
 * 新增的 migrateSchemaDueTime()，不需要重跑本文件的 setupSheets()；
 * 全新建表才会用到这里的改动。
 *
 * 【V4.6 新增】createTriggers() 新增 triggerDailyStatisticsRecompute
 * 每日触发器——TaskStatistics 不再由 10_ProjectionEngine.gs 同步增量
 * 维护，改成每天批量重算一次，见 00_ADR.gs ADR-2026-07-06-005。已经跑过
 * 旧版 createTriggers() 的部署，需要重新跑一次这个函数才能补上新触发器
 * （createTriggers() 是幂等的，会先清掉同名旧触发器再重新创建，可以放心
 * 重复运行）。
 *
 * 【V4.3 修复】setupSheets() 之前完全没有创建 Events 表，日志里还写着
 * "Events 由 Core 项目建"——这个假设在"先跑过 Core 项目的 setupSheets()
 * 或等价初始化"的前提下成立，但如果有人（比如本项目自己独立诊断/测试时）
 * 先对着一张全新的 Spreadsheet 只跑本项目的 setupSheets()，Events 表根本
 * 不存在，后续任何 createTask 都会在 02_EventBus._sheet_() 里直接抛错
 * "Events sheet 不存在"（实测复现：runDiagnostics → createTask →
 * IdempotencyManager.createTaskIfNotExists → TaskEngine.createTaskDirect_ →
 * EventBus.publish → _sheet_ 抛错）。
 *
 * 修复方式：setupSheets() 现在也会确保 Events 表存在（表头跟
 * 02_EventBus.gs 的 COLS 数组完全一致：event_id/timestamp/type/chat_id/
 * payload/source）。这不违反"Events 是跟 Core/Reminder OS 共享的表"这个
 * 架构事实——_ensureSheet_() 只在表完全是空的（getLastRow()===0）时才写
 * 表头，如果 Core 项目已经先建好了 Events 表（哪怕只有表头没有数据），
 * 本项目的 setupSheets() 对它是完全无操作的只读探测，不会覆盖或弄乱
 * 已有数据；反过来，如果本项目先跑，也能正确把 Events 表建起来，不需要
 * 强制要求"必须先跑 Core 项目的初始化"这个前置顺序。repairSheetHeaders()
 * 和 runDiagnostics() 的检查列表同步加入了 Events。
 *
 * 【V4.2 审计修复】MEDIUM RISK 3：_repairOneSheetHeader_ 原来只比对第一
 * 格是否等于 headers[0]，改成对整行做完全比对（见函数头注释）。
 *
 * 【V4 变更】
 *   1. Tasks/ActiveTasks/ArchiveTasks 表头新增 description/tags 两列
 *      （向后兼容追加列，不影响旧数据）。
 *   2. 新增 TaskStatistics / TaskFilters 两张 Projection 表。
 *   3. runDiagnostics 覆盖新表 + 新 Engine 的基本冒烟测试。
 *
 * 使用顺序（重要，顺序不能反）：
 *   1. 先确认本项目 Script Properties 里的 SPREADSHEET_ID 已经设置成跟
 *      Personal AI Core 项目【同一张】Spreadsheet 的 ID。
 *   2. 跑一次 setupSheets() —— 建好全部六张表（幂等，已存在的表/列不会
 *      被破坏；Events 表如果 Core 项目已经建过，这里对它只读探测不改动）。
 *   3. （仅从 V3 或更早版本迁移时需要）跑一次 migrateSchemaV4()（在
 *      11_ProjectionRebuilder.gs 里）。
 *   4. 本项目【不需要】单独 Deploy as Web App——它是被 Personal AI Core
 *      当 Library 调用的。
 *   5. 跑一次 createTriggers() —— 挂上本项目自己的定时任务。
 *   6.（仅首次上线/迁移、Events 已有历史 Task 事件时需要）跑一次
 *      rebuildAllProjections()（在 11_ProjectionRebuilder.gs 里）。
 */

// ============ 1. 建表 ============

function setupSheets() {
  var id = SecureConfig.getKey('SPREADSHEET_ID');
  if (!id) {
    Logger.log('❌ 先设置 SPREADSHEET_ID: SecureConfig.setKey("SPREADSHEET_ID", "跟Core项目一样的Spreadsheet ID")');
    return;
  }
  var ss = SpreadsheetApp.openById(id);

  // 【V4.3 修复】Events 表之前被 setupSheets() 完全忽略——见文件头修复说明。
  // 表头必须跟 02_EventBus.gs 的 COLS 数组逐字一致，这是本项目/Core 项目/
  // Reminder OS 三方共享同一张 Events 表时唯一需要保持同步的地方。
  _ensureSheet_(ss, 'Events', [
    'event_id', 'timestamp', 'type', 'chat_id', 'payload', 'source'
  ]);

  _ensureSheet_(ss, 'Tasks', [
    'task_id', 'timestamp', 'title', 'category', 'status', 'due_date',
    'due_time', 'due_datetime', 'recurring', 'priority', 'context', 'budget',
    'notes', 'description', 'tags', 'chat_id', 'completed_at',
    'reminder_count', 'identity', 'archived'
  ]);

  _ensureSheet_(ss, 'ActiveTasks', [
    'task_id', 'timestamp', 'title', 'category', 'status', 'due_date',
    'due_time', 'due_datetime', 'recurring', 'priority', 'context', 'budget',
    'notes', 'description', 'tags', 'chat_id', 'completed_at',
    'reminder_count', 'identity'
  ]);

  _ensureSheet_(ss, 'ArchiveTasks', [
    'task_id', 'timestamp', 'title', 'category', 'status', 'due_date',
    'due_time', 'due_datetime', 'recurring', 'priority', 'context', 'budget',
    'notes', 'description', 'tags', 'chat_id', 'completed_at',
    'reminder_count', 'identity', 'archived_at'
  ]);

  // V4 新增：TaskStatistics（每 chat_id 一行的累计计数器，10_ProjectionEngine 增量维护）
  _ensureSheet_(ss, 'TaskStatistics', [
    'chat_id', 'total_count', 'pending_count', 'done_count', 'cancelled_count',
    'recurring_count', 'reminder_count_total', 'last_updated_at'
  ]);

  // V4 新增：TaskFilters（每 task_id 一行的搜索用扁平投影）
  _ensureSheet_(ss, 'TaskFilters', [
    'task_id', 'chat_id', 'searchable_text', 'tags_csv'
  ]);

  Logger.log('✅ Sheets 就位: Events, Tasks, ActiveTasks, ArchiveTasks, TaskStatistics, TaskFilters');
  Logger.log('   （Events 表如果 Core 项目已经建过，上面这行只是确认存在，不会动它的数据）');
  Logger.log('下一步: 如果 Tasks 已有旧数据（V3 或更早） → migrateSchemaV4()；然后 createTriggers()');
}

function _ensureSheet_(ss, name, headers) {
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
  }
  if (sheet.getLastRow() === 0) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.setFrozenRows(1);
    sheet.getRange(2, 1, Math.max(sheet.getMaxRows() - 1, 1), headers.length).setNumberFormat('@');
  }
  return sheet;
}

// ============ 2. 定时触发器 ============

/**
 * 本项目现在有两个定时任务：
 *   triggerDailyArchive              — 每天凌晨2点（冷归档，13_ActiveTasksEngine.gs）
 *   triggerDailyStatisticsRecompute  — 每天凌晨3点（V4.6新增，TaskStatistics
 *                                       批量重算，11_ProjectionRebuilder.gs）——
 *                                       错开1小时跑，避免两个每日任务同一
 *                                       时刻抢 Sheet 资源（虽然两者操作的
 *                                       是不同表，理论上不冲突，错开纯粹是
 *                                       图个干净，没有实质必要性）。
 * TaskStatistics/TaskFilters 都是事件驱动或按需生成的，V4.6 前两者都不需要
 * 定时任务；V4.6 起 TaskStatistics 改为每日批量重算（见 00_ADR.gs
 * ADR-2026-07-06-005），需要这第二个触发器。如果发现漂移，也可以手动跑
 * 11_ProjectionRebuilder.gs 的对应 rebuild/recompute 函数。
 */
function createTriggers() {
  var handlerNames = ['triggerDailyArchive', 'triggerDailyStatisticsRecompute'];
  ScriptApp.getProjectTriggers().forEach(function (t) {
    if (handlerNames.indexOf(t.getHandlerFunction()) !== -1) {
      ScriptApp.deleteTrigger(t);
    }
  });

  ScriptApp.newTrigger('triggerDailyArchive').timeBased().everyDays(1).atHour(2).create();
  ScriptApp.newTrigger('triggerDailyStatisticsRecompute').timeBased().everyDays(1).atHour(3).create();

  Logger.log('✅ Productivity OS 自己的触发器挂好了:');
  Logger.log('  triggerDailyArchive — 每天 02:00 左右（冷归档）');
  Logger.log('  triggerDailyStatisticsRecompute — 每天 03:00 左右（TaskStatistics 批量重算，V4.6新增）');
}

/**
 * 【V4.2 修复 MEDIUM RISK 3】原来只比对第一行第一格（'task_id'/'chat_id'）
 * 是否等于 headers[0]，如果用户不小心删掉右侧某一列（比如 identity/tags/
 * description）或者把列顺序打乱，第一格照样对得上，函数会误判"表头正确"
 * 直接跳过——缺失的列永远不会被自动补全，直到某次写入因为找不到对应
 * 表头列而报错崩溃才会被发现。
 *
 * 现在改成对第一行整行数组（长度取 headers.length 和当前表实际列数的
 * 较大值，这样右侧被删掉的列也能被检测出来）跟标准 headers 数组做逐个
 * 元素 + 顺序的完全比对，只有完全一致才跳过。
 *
 * 两种不一致场景处理方式不同：
 *  (a) 第一格还是对的（这行"看起来"仍然是表头行，只是右侧列缺失/顺序
 *      错了/被改名）→ 原地覆写第一行为标准 headers，不触碰下面任何数据行。
 *  (b) 连第一格都不对（第一行已经变成一行真实数据，说明表头行本身被
 *      删掉了）→ 沿用原来的做法：插入一行新表头，原有数据整体下移一行。
 */
function _repairOneSheetHeader_(name, headers) {
  var id = SecureConfig.getKey('SPREADSHEET_ID');
  var ss = SpreadsheetApp.openById(id);
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    Logger.log('❌ ' + name + ' 不存在，先跑 setupSheets()');
    return;
  }

  var currentWidth = Math.max(sheet.getLastColumn(), headers.length);
  var currentRow = (currentWidth > 0)
    ? sheet.getRange(1, 1, 1, currentWidth).getValues()[0].map(function (v) { return String(v).trim(); })
    : [];

  var isExactMatch = headers.every(function (h, idx) { return currentRow[idx] === h; }) &&
    currentRow.slice(headers.length).every(function (v) { return v === ''; });

  if (isExactMatch) {
    Logger.log('✅ ' + name + ' 表头完整且顺序正确（全部 ' + headers.length + ' 列核对通过），跳过');
    return;
  }

  var firstCell = currentRow[0] || '';
  var looksLikeHeaderRow = (firstCell === headers[0]);

  if (looksLikeHeaderRow) {
    var missing = headers.filter(function (h) { return currentRow.indexOf(h) === -1; });
    Logger.log('⚠️ ' + name + ' 表头列不完整或顺序不对（当前第一行: ' + JSON.stringify(currentRow) +
      '），缺失/需修正的列: ' + JSON.stringify(missing) + '。原地覆写第一行为标准表头（不触碰数据行）。');
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.setFrozenRows(1);
    Logger.log('✅ ' + name + ' 表头列已修复');
    return;
  }

  Logger.log('⚠️ ' + name + ' 第一行不像表头（第一格="' + firstCell + '"，应为"' + headers[0] +
    '"），说明表头行本身被删掉了。插入新的表头行，原有数据整体往下挪一行');
  sheet.insertRowBefore(1);
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  sheet.setFrozenRows(1);
  sheet.getRange(2, 1, Math.max(sheet.getMaxRows() - 1, 1), headers.length).setNumberFormat('@');
  Logger.log('✅ ' + name + ' 表头已修复（插入新行）');
}

// ============ 诊断：不碰 Telegram，直接看本项目状态 ============

function runDiagnostics() {
  Logger.log('========== Productivity OS V4 诊断开始 ==========');

  var id = SecureConfig.getKey('SPREADSHEET_ID');
  if (!id) {
    Logger.log('❌ SPREADSHEET_ID 没设置，下面的检查会全部失败');
  }
  var ss = id ? SpreadsheetApp.openById(id) : null;

  ['Events', 'Tasks', 'ActiveTasks', 'ArchiveTasks', 'TaskStatistics', 'TaskFilters'].forEach(function (name) {
    if (!ss) return;
    var sheet = ss.getSheetByName(name);
    if (!sheet) {
      Logger.log('❌ Sheet "' + name + '" 不存在！先跑 setupSheets()' +
        (name === 'Events' ? '（V4.3 起 setupSheets() 也会建 Events 表了，见文件头修复说明）' : ''));
    } else {
      Logger.log('✅ Sheet "' + name + '" 存在，表头: ' + JSON.stringify(sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]));
      Logger.log('   当前行数（含表头）: ' + sheet.getLastRow());
    }
  });

  var testChatId = 'diagnostic_test_chat';

  try {
    var parsed = parseTaskIntent('提醒我明天3点交房租');
    Logger.log('✅ parseTaskIntent 测试结果: ' + JSON.stringify(parsed));
  } catch (e) {
    Logger.log('❌ parseTaskIntent 本身就报错: ' + e.message + '\n' + e.stack);
  }

  var testTaskId = null;
  try {
    var task = createTask('诊断测试任务-可以删除', { category: 'GENERAL', notes: 'DIAGNOSTIC_TEST_TEMP' }, testChatId);
    testTaskId = task.task_id;
    Logger.log('✅ createTask 测试结果: ' + JSON.stringify(task));
  } catch (e) {
    Logger.log('❌ createTask 报错: ' + e.message + '\n' + e.stack);
  }

  if (testTaskId) {
    try {
      var updated = updateTask(testTaskId, { notes: 'DIAGNOSTIC_TEST_TEMP_UPDATED' }, testChatId);
      Logger.log('✅ updateTask 测试结果: ' + JSON.stringify(updated));
    } catch (e) {
      Logger.log('❌ updateTask 报错: ' + e.message + '\n' + e.stack);
    }

    try {
      var pending = TaskQueryEngine.getPendingTasks(testChatId);
      Logger.log('✅ TaskQueryEngine.getPendingTasks 测试结果，共 ' + pending.length + ' 条');
    } catch (e) {
      Logger.log('❌ TaskQueryEngine.getPendingTasks 报错: ' + e.message + '\n' + e.stack);
    }

    try {
      var dash = TaskQueryEngine.getDashboard('today', testChatId);
      Logger.log('✅ TaskQueryEngine.getDashboard 测试结果:\n' + dash);
    } catch (e) {
      Logger.log('❌ TaskQueryEngine.getDashboard 报错: ' + e.message + '\n' + e.stack);
    }

    try {
      cancelTask(testTaskId, testChatId);
      Logger.log('✅ 已自动取消诊断测试任务');
    } catch (e) {
      Logger.log('❌ cancelTask 报错: ' + e.message + '\n' + e.stack);
    }
  }

  Logger.log('⚠️ 这些是本项目内部测试，不代表 Core 项目能成功调用本项目（Library 引入是否');
  Logger.log('   配置正确要去 Core 项目那边测：ProductivityOS.handleTaskIntent(...)）。');
  Logger.log('========== 诊断结束 ==========');
}
