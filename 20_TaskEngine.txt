/**
 * 20_TaskEngine.gs
 * Productivity OS v4.2 — Task Engine（create / update / complete / cancel）
 *
 * 【V4.2 审计修复】（2026-07-06 第二轮外部审计，完整记录见
 * 00_Project_State.gs"已知Bug"）：
 *
 *  - HIGH RISK 2 修复：updateTask 修改 title/due_date/category/priority/
 *    recurring 任一字段时，重新计算 identity 并写回，避免去重系统认旧值。
 *  - HIGH RISK 3 修复：completeTask/cancelTask/updateTask 不再无条件调用
 *    materializeTaskRow_ 安全兜底——EventBus.publish() 内部已经同步调用
 *    ProjectionEngine.dispatch() 并在失败时返回 event.projection_ok=false，
 *    只有这个标志为 false（真的失败了）才触发一次额外写入，正常路径下
 *    每次操作只有 Projection 那一次 Sheet I/O，不再有"写两遍"的冗余。
 *  - MEDIUM RISK 1 修复：completeTask/cancelTask 在发布事件前先校验任务
 *    存在，任务不存在（taskId 打错/已删）时直接拦截返回，不再产生
 *    title/chat_id 全为空的"幽灵任务行"。
 *  - LOW RISK 1 修复：删除文件尾部裸全局变量声明
 *    （TASKS_SHEET_NAME/TASK_CATEGORIES/TASK_PRIORITIES/TASK_RECURRING），
 *    全部改成读 ProductivityConfig.xxx。这几个名字足够通用，一旦 Core
 *    主项目或其他 Library 恰好用了同名全局变量会被静默覆盖，风险面比
 *    "少打几个字"划算不来。
 *
 * 【V4 改名+重构】原 20_ProductivityModule.gs：
 *
 * 1. 【修复 LOW RISK 2（V4初版审计）】整个模块包进 IIFE。原来的裸全局函数
 *    _createTaskDirect_ 变成 TaskEngine.createTaskDirect_（保留下划线
 *    后缀标注"内部函数，设计上只允许 09_IdempotencyManager 调用"的语义，
 *    但物理上不再是可以被任何文件意外碰到的裸全局）。
 *    createTask/completeTask/cancelTask/getPendingTasks 保留同名全局函数
 *    作为向后兼容 wrapper，委托给 TaskEngine 的 IIFE 方法——Core 项目
 *    04_Main.gs 里既有的 ProductivityOS.createTask(...) 这类 Library 调用
 *    不用改一行代码（00_Project_Constitution.gs 要求"维持向后兼容"）。
 *
 * 2. 新增 updateTask()——V3 完全没有"更新任务"的能力，只有 create/complete/
 *    cancel。V4 规格明确要求 Task Engine 是 create/update/complete/cancel
 *    四件套，这是新功能补齐，不是bug修复。新增 TASK_UPDATED 事件类型。
 *
 * 3. recurring 续期逻辑（原 _spawnNextRecurrenceIfNeeded_）搬到
 *    21_RecurringEngine.gs，completeTask() 改成调用
 *    RecurringEngine.spawnNextIfNeeded()——职责分离：TaskEngine 只管
 *    "完成这一次"的事实写入，"要不要生成下一次"是 Intelligence 层
 *    （Recurring 属于 Blueprint 里 Runtime 的 Execution 延伸，但逻辑本身
 *    独立成 Engine 更符合"一个 Engine 只做一件事"）。
 *
 * 架构铁律（不变）：
 *  - 真相来源是 EVENTS 表
 *  - Tasks Sheet 是 Read Model（由 10_ProjectionEngine 维护）
 *  - 只有 EventBus.publish 能写 Events
 *  - createTaskDirect_ 只允许 09_IdempotencyManager 调用，外部代码一律走
 *    createTask()（内部会经过幂等+锁）
 */

var ProductivityConfig = Object.freeze({
  TASKS_SHEET_NAME: 'Tasks',
  TASK_CATEGORIES:  ['MAINTENANCE', 'SHOPPING', 'ADMIN', 'HEALTH', 'GENERAL', 'PROJECT'],
  TASK_PRIORITIES:  ['LOW', 'MEDIUM', 'HIGH', 'CRITICAL'],
  TASK_RECURRING:   ['', 'Daily', 'Weekly', 'Monthly', 'Yearly'],
  // HIGH RISK 2 修复用：这几个字段任一变化都会让 IdentityEngine 算出不同
  // 的 identity 哈希，updateTask 改到其中任一个都必须重新计算 identity。
  IDENTITY_AFFECTING_FIELDS: ['title', 'due_date', 'recurring', 'priority', 'category']
});

var TaskEngine = (function () {

  var CFG = ProductivityConfig; // 内部简写，不额外暴露成新的裸全局

  // ============ Create ============

  function createTask(title, meta, chatId) {
    var result = IdempotencyManager.createTaskIfNotExists(title, meta || {}, chatId);
    return result.task;
  }

  /**
   * 实际创建函数 —— 只由 09_IdempotencyManager.createTaskIfNotExists() 调用
   * （已在锁内）。外部代码一律调 createTask()；禁止绕过 IdempotencyManager
   * 直接调本函数。
   */
  function createTaskDirect_(title, meta, chatId, identity) {
    meta = meta || {};

    var category  = CFG.TASK_CATEGORIES.indexOf(meta.category) !== -1 ? meta.category : 'GENERAL';
    var priority   = CFG.TASK_PRIORITIES.indexOf(meta.priority) !== -1 ? meta.priority : 'MEDIUM';
    var recurring  = CFG.TASK_RECURRING.indexOf(meta.recurring) !== -1 ? meta.recurring : '';

    var task = {
      task_id:        generateTaskId_(),
      identity:       identity || '',
      timestamp:      new Date().toISOString(),
      title:          title,
      category:       category,
      status:         'PENDING',
      due_date:       meta.due_date     || '',
      recurring:      recurring,
      priority:       priority,
      context:        meta.context      || '',
      budget:         (meta.budget != null) ? meta.budget : '',
      notes:          meta.notes        || '',
      description:    meta.description  || '',
      tags:           meta.tags          || '',
      chat_id:        chatId             || '',
      completed_at:   '',
      reminder_count: 0,
      archived:       false
    };

    // EventBus.publish() 会自动调 ProjectionEngine.dispatch() 写
    // Tasks + ActiveTasks + TaskFilters。创建路径本来就没有额外的安全兜底
    // 写入（只有 update/complete/cancel 曾经有，且已在 V4.2 改成"仅失败时"）。
    EventBus.publish('TASK_CREATED', task, chatId, 'TaskEngine', identity);

    return task;
  }

  // ============ Update（V4新增，V4.2 修复 identity 未重算） ============

  /**
   * 更新任务的可编辑字段。
   *
   * 允许更新的字段白名单：title, category, priority, due_date, recurring,
   * context, budget, notes, description, tags（不允许改 task_id/chat_id/
   * status/completed_at/identity——identity 是本函数内部按需自动重算的
   * 派生值，不接受外部直接传入覆盖）。
   *
   * 【HIGH RISK 2 修复】DeduplicationEngine.findExistingTask 完全依赖
   * identity 列。title/due_date/category/priority/recurring 任一个变了，
   * IdentityEngine 算出来的哈希就会不同——之前 updateTask 改了这些字段但
   * 没有同步重算 identity，导致 Sheet 里留着旧哈希：旧属性的新任务会被
   * 误判"已存在"而拦截，新属性的重复任务却因为哈希对不上而漏判重复。
   * 现在只要 payload 命中 CFG.IDENTITY_AFFECTING_FIELDS 里任一个，就用
   * "已有字段 + 本次改动"合并后的完整值重新算一次 identity，一并写进
   * payload（TASK_UPDATED 事件 + Sheet）。
   *
   * @param {string} taskId
   * @param {object} changes
   * @param {string} chatId
   * @returns {object|null}
   */
  var UPDATABLE_FIELDS = ['title', 'category', 'priority', 'due_date', 'recurring',
                           'context', 'budget', 'notes', 'description', 'tags'];

  function updateTask(taskId, changes, chatId) {
    var existing = TaskQueryEngine.getTask(taskId, chatId);
    if (!existing) {
      Logger.log('[TaskEngine] updateTask: 找不到任务 ' + taskId + '，拦截（防止产生幽灵行）');
      return null;
    }

    changes = changes || {};
    var payload = { task_id: taskId };
    UPDATABLE_FIELDS.forEach(function (f) {
      if (changes.hasOwnProperty(f)) {
        var v = changes[f];
        if (f === 'category'  && CFG.TASK_CATEGORIES.indexOf(v) === -1) return;
        if (f === 'priority'  && CFG.TASK_PRIORITIES.indexOf(v) === -1) return;
        if (f === 'recurring' && CFG.TASK_RECURRING.indexOf(v) === -1) return;
        payload[f] = v;
      }
    });

    if (Object.keys(payload).length === 1) {
      return null; // 只有 task_id，没有任何合法字段被改——不发空事件
    }

    // 合并"已有字段 + 本次改动"，供下面 identity 重算和失败兜底共用，
    // 避免同一份数据算两次。
    var merged = shallowCopy_(existing);
    for (var k in payload) merged[k] = payload[k];

    var identityFieldChanged = CFG.IDENTITY_AFFECTING_FIELDS.some(function (f) {
      return payload.hasOwnProperty(f);
    });
    if (identityFieldChanged) {
      var newIdentity = IdentityEngine.generateTaskIdentity(
        merged.chat_id || chatId || existing.chat_id,
        merged.title,
        merged.due_date  || '',
        merged.recurring || '',
        merged.priority  || 'MEDIUM',
        merged.category  || 'GENERAL'
      );
      payload.identity = newIdentity;
      merged.identity   = newIdentity;
    }

    var event = EventBus.publish('TASK_UPDATED', payload, chatId || existing.chat_id, 'TaskEngine');

    // 【HIGH RISK 3 修复】只有 Projection 真的失败了才做安全兜底写入
    // （event.projection_ok 由 02_EventBus.gs 在 dispatch 失败时置 false）。
    if (event && event.projection_ok === false) {
      materializeTaskRow_(taskId, merged);
    }

    return payload;
  }

  // ============ Complete ============

  /**
   * 标记任务完成。
   *
   * 【MEDIUM RISK 1 修复】先用 TaskQueryEngine.getTask 校验任务存在；
   * 找不到直接拦截返回，不发布事件——避免 materializeTaskRow_ 在 Tasks
   * 表追加一行除 status/completed_at 外全为空的"幽灵行"。
   *
   * @returns {{next_task: (object|null), not_found: (boolean|undefined)}}
   */
  function completeTask(taskId, chatId) {
    var taskBeforeComplete = TaskQueryEngine.getTask(taskId, chatId);
    if (!taskBeforeComplete) {
      Logger.log('[TaskEngine] completeTask: 找不到任务 ' + taskId + '，拦截（防止产生幽灵行）');
      return { next_task: null, not_found: true };
    }

    var event = EventBus.publish('TASK_COMPLETED', { task_id: taskId }, chatId, 'TaskEngine');

    // 【HIGH RISK 3 修复】只有 Projection 真的失败才兜底写一次
    if (event && event.projection_ok === false) {
      materializeTaskRow_(taskId, { status: 'DONE', completed_at: event.timestamp });
    }

    var nextTask = RecurringEngine.spawnNextIfNeeded(taskBeforeComplete, chatId);
    return { next_task: nextTask };
  }

  /**
   * 取消任务。【MEDIUM RISK 1 修复】同 completeTask，先校验任务存在。
   * @returns {{not_found: (boolean|undefined)}}
   */
  function cancelTask(taskId, chatId) {
    var existing = TaskQueryEngine.getTask(taskId, chatId);
    if (!existing) {
      Logger.log('[TaskEngine] cancelTask: 找不到任务 ' + taskId + '，拦截（防止产生幽灵行）');
      return { not_found: true };
    }

    var event = EventBus.publish('TASK_CANCELLED', { task_id: taskId }, chatId, 'TaskEngine');

    if (event && event.projection_ok === false) {
      materializeTaskRow_(taskId, { status: 'CANCELLED' });
    }

    return {};
  }

  function getPendingTasks(chatId) {
    return TaskQueryEngine.getTasks(chatId, { status: 'PENDING' });
  }

  // ============ 派生引擎（保留供 11_ProjectionRebuilder 使用） ============

  function deriveFromEvent(event, stateMap) {
    stateMap = stateMap || {};
    var p = event.payload || {};

    switch (event.type) {
      case 'TASK_CREATED':
        stateMap[p.task_id] = shallowCopy_(p);
        if (!stateMap[p.task_id].chat_id) stateMap[p.task_id].chat_id = event.chat_id;
        break;

      case 'TASK_UPDATED':
        if (stateMap[p.task_id]) {
          for (var k in p) {
            if (k !== 'task_id') stateMap[p.task_id][k] = p[k];
          }
        }
        break;

      case 'TASK_COMPLETED':
        if (stateMap[p.task_id]) {
          stateMap[p.task_id].status       = 'DONE';
          stateMap[p.task_id].completed_at = event.timestamp;
        }
        break;

      case 'TASK_CANCELLED':
        if (stateMap[p.task_id]) {
          stateMap[p.task_id].status = 'CANCELLED';
        }
        break;

      case 'REMINDER_SENT':
        if (stateMap[p.task_id]) {
          stateMap[p.task_id].reminder_count = (stateMap[p.task_id].reminder_count || 0) + 1;
          stateMap[p.task_id].last_reminder_at = p.sent_at || event.timestamp;
        }
        break;
    }

    return stateMap;
  }

  function deriveTaskState_() {
    var events = EventBus.getAllEvents();
    var state = {};
    events.forEach(function (e) {
      deriveFromEvent(e, state);
    });
    return state;
  }

  // ============ 派生视图维护（Read Model 辅助） ============

  /**
   * 【V4.2 起调用时机收紧】仅在以下场景调用：
   *   (a) updateTask/completeTask/cancelTask 检测到 event.projection_ok===false
   *       时的失败兜底
   *   (b) rebuildTasksSheet_()（手动全量重建快捷函数）
   */
  function materializeTaskRow_(taskId, knownTask) {
    var task = knownTask;
    if (!task) {
      task = deriveTaskState_()[taskId];
    }
    if (!task) return;

    upsertRowByKey_(CFG.TASKS_SHEET_NAME, 'task_id', taskId, task);

    var finalStatus = String(task.status || '').toUpperCase();
    if (finalStatus === 'DONE' || finalStatus === 'CANCELLED') {
      try {
        deleteRowByKey_('ActiveTasks', 'task_id', taskId);
      } catch (e) {
        Logger.log('[TaskEngine] ActiveTasks 安全网清理失败（不影响主流程，Tasks已正确更新）: ' + e.message);
      }
    } else {
      try {
        upsertRowByKey_('ActiveTasks', 'task_id', taskId, task);
      } catch (e) {
        Logger.log('[TaskEngine] ActiveTasks 安全网写入失败: ' + e.message);
      }
    }
  }

  function rebuildTasksSheet_() {
    rebuildTasksProjection(); // 11_ProjectionRebuilder.gs
  }

  // ============ 内部工具 ============

  function generateTaskId_() {
    var tz = Session.getScriptTimeZone();
    var today = Utilities.formatDate(new Date(), tz, 'yyyyMMdd');
    var uniqueSuffix = Utilities.getUuid().split('-')[0].toUpperCase();
    return 'TSK-' + today + '-' + uniqueSuffix;
  }

  return {
    createTask:         createTask,
    createTaskDirect_:  createTaskDirect_,
    updateTask:         updateTask,
    completeTask:        completeTask,
    cancelTask:          cancelTask,
    getPendingTasks:     getPendingTasks,
    deriveFromEvent:     deriveFromEvent,
    deriveTaskState_:    deriveTaskState_,
    materializeTaskRow_: materializeTaskRow_,
    rebuildTasksSheet_:  rebuildTasksSheet_
  };
})();

  return TaskEngine.deriveFromEvent(event, stateMap);
}
