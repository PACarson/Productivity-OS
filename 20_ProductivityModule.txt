  TASKS_SHEET_NAME: 'Tasks',
  TASK_CATEGORIES:  ['MAINTENANCE', 'SHOPPING', 'ADMIN', 'HEALTH', 'GENERAL', 'PROJECT'],
  TASK_PRIORITIES:  ['LOW', 'MEDIUM', 'HIGH', 'CRITICAL'],
  TASK_RECURRING:   ['', 'Daily', 'Weekly', 'Monthly', 'Yearly']
});

// 向后兼容别名（指向同一个数组引用，不是拷贝）—— 现有调用代码不用改
var TASKS_SHEET_NAME = ProductivityConfig.TASKS_SHEET_NAME;
var TASK_CATEGORIES  = ProductivityConfig.TASK_CATEGORIES;
var TASK_PRIORITIES  = ProductivityConfig.TASK_PRIORITIES;
var TASK_RECURRING   = ProductivityConfig.TASK_RECURRING;

// ============ 对外函数 ============

/**
 * 创建任务 —— v3.0：幂等路由层（内部已有 ScriptLock，见 09_IdempotencyManager）
 *
 * @param {string} title
 * @param {object} meta  { category, priority, due_date, recurring, context, budget, notes }
 * @param {string} chatId
 * @returns {object} task 对象（新建或已有）
 */
function createTask(title, meta, chatId) {
  var result = IdempotencyManager.createTaskIfNotExists(title, meta || {}, chatId);
  return result.task;
}

/**
 * 实际创建函数 —— 只由 09_IdempotencyManager.createTaskIfNotExists() 调用（已在锁内）。
 * 外部代码一律调 createTask()；禁止绕过 IdempotencyManager 直接调本函数。
 *
 * @param {string} title
 * @param {object} meta
 * @param {string} chatId
 * @param {string} identity
 * @returns {object} task
 */
function _createTaskDirect_(title, meta, chatId, identity) {
  meta = meta || {};

  var category  = TASK_CATEGORIES.indexOf(meta.category) !== -1 ? meta.category : 'GENERAL';
  var priority   = TASK_PRIORITIES.indexOf(meta.priority) !== -1 ? meta.priority : 'MEDIUM';
  var recurring  = TASK_RECURRING.indexOf(meta.recurring) !== -1 ? meta.recurring : '';

  var task = {
    task_id:        generateTaskId_(),
    identity:       identity || '',
    timestamp:      new Date().toISOString(),
    title:          title,
    category:       category,
    status:         'PENDING',
    due_date:       meta.due_date  || '',
    recurring:      recurring,
    priority:       priority,
    context:        meta.context   || '',
    budget:         (meta.budget  != null) ? meta.budget : '',
    notes:          meta.notes     || '',
    chat_id:        chatId         || '',
    completed_at:   '',
    reminder_count: 0
  };

  // EventBus.publish() 会自动调 ProjectionEngine.dispatch() 写 Tasks + ActiveTasks
  EventBus.publish('TASK_CREATED', task, chatId, 'ProductivityModule', identity);

  return task;
}

/**
 * 标记任务完成
 *
 * v3.2 新增（回应"每月25号交房租done了，没有自动生成下个月的task"）：
 * 如果被完成的任务本身是recurring的（Daily/Weekly/Monthly/Yearly之一），
 * 完成的同时自动创建下一次实例（due_date按同一周期推进一格）。
 *
 * 实现细节见 _spawnNextRecurrenceIfNeeded_ 和 09_TemporalParser.gs 的
 * computeNextDueDateFromLabel；范围只到interval=1的四种legacy类型，
 * 跟现有TASK_RECURRING能持久化的范围一致，不是新引入的限制。
 *
 * 只对completeTask生效，cancelTask不会自动生成下一次——取消是用户主动
 * 表示"这一次不需要了"，是否要连未来所有次都停掉是有歧义的产品行为
 * （比如"每年生日"被取消一次，明年该不该继续提醒？），这次不擅自决定，
 * 只解决被明确报告的"完成后没续上"这一种情况。
 *
 * @returns {{next_task: (object|null)}}  next_task非recurring或生成失败时为null
 */
function completeTask(taskId, chatId) {
  var taskBeforeComplete = QueryEngine.getTaskById(taskId); // 完成前先取快照，拿title/recurring/due_date

  EventBus.publish('TASK_COMPLETED', { task_id: taskId }, chatId, 'ProductivityModule');
  // 安全兜底：ProjectionEngine 失败时确保 Tasks（以及 v3.1 新增的 ActiveTasks）仍被更新
  _materializeTaskRow_(taskId, { status: 'DONE', completed_at: new Date().toISOString() });

  var nextTask = _spawnNextRecurrenceIfNeeded_(taskBeforeComplete, chatId);
  return { next_task: nextTask };
}

/**
 * 如果被完成的任务本身是recurring的，生成下一次实例；否则返回null。
 *
 * 静默失败（比如taskBeforeComplete没取到、due_date解析不出来、
 * IdempotencyManager抛错）不影响completeTask主流程——"这次完成"本身必须
 * 成功，"下一次要不要生出来"是锦上添花，不能反过来把主流程搞挂。
 *
 * 幂等性说明：不需要单独加锁/去重保护。走的是
 * IdempotencyManager.createTaskIfNotExists同一条路径，算出来的
 * identity=hash(chat_id+title+due_date+recurring+priority+category)——
 * 只要taskBeforeComplete.due_date没变，两次调用（比如webhook重试导致
 * completeTask被调两次）算出的nextDueDate完全一样，DeduplicationEngine
 * 会把第二次识别成"已存在"直接跳过。
 *
 * @param {object} task  完成前的task快照（QueryEngine.getTaskById的返回值）
 * @param {string} chatId
 * @returns {object|null}
 */
function _spawnNextRecurrenceIfNeeded_(task, chatId) {
  if (!task || !task.recurring || !task.due_date) return null;

  try {
    var nextDueDate = computeNextDueDateFromLabel(task.due_date, task.recurring); // 09_TemporalParser.gs
    if (!nextDueDate) return null;

    var result = IdempotencyManager.createTaskIfNotExists(
      task.title,
      {
        due_date:  nextDueDate,
        recurring: task.recurring,
        category:  task.category,
        priority:  task.priority,
        context:   task.context,
        budget:    task.budget,
        notes:     task.notes
      },
      chatId || task.chat_id
    );
    return result.task;
  } catch (e) {
    Logger.log('[ProductivityModule] _spawnNextRecurrenceIfNeeded_ 失败（不影响本次completeTask主流程）: ' + e.message);
    return null;
  }
}

/**
 * 取消任务
 */
function cancelTask(taskId, chatId) {
  EventBus.publish('TASK_CANCELLED', { task_id: taskId }, chatId, 'ProductivityModule');
  _materializeTaskRow_(taskId, { status: 'CANCELLED' });
}

// ============ 派生引擎（保留供 11_ProjectionRebuilder 使用） ============

/**
 * 纯函数 reducer：给定当前状态 map 和一个事件，返回更新后的状态 map。
 */
function deriveFromEvent(event, stateMap) {
  stateMap = stateMap || {};
  var p = event.payload || {};

  switch (event.type) {
    case 'TASK_CREATED':
      stateMap[p.task_id] = shallowCopy_(p); // v3.1：05_SheetUtils.gs 共用版本
      if (!stateMap[p.task_id].chat_id) stateMap[p.task_id].chat_id = event.chat_id;
      break;

    case 'TASK_COMPLETED':
      if (stateMap[p.task_id]) {
        stateMap[p.task_id].status       = 'DONE';
        stateMap[p.task_id].completed_at = event.timestamp;
      }
      break;

    case 'TASK_CANCELLED':
      if (stateMap[p.task_id]) {
        stateMap[p.task_id].status = 'CANCELLED';
      }
      break;

    case 'REMINDER_SENT':
      if (stateMap[p.task_id]) {
        stateMap[p.task_id].reminder_count = (stateMap[p.task_id].reminder_count || 0) + 1;
        stateMap[p.task_id].last_reminder_at = p.sent_at || event.timestamp;
      }
      break;
  }

  return stateMap;
}

/**
 * 全量重放 Events → Task state map。
 * 正常路径不调用此函数；只供 11_ProjectionRebuilder 和调试使用。
 */
function deriveTaskState_() {
  var events = EventBus.getAllEvents();
  var state = {};
  events.forEach(function (e) {
    deriveFromEvent(e, state);
  });
  return state;
}

// ============ 派生视图维护（Read Model 辅助） ============

/**
 * 把某个 task 当前算出来的状态覆写回 Tasks Sheet 对应行。
 *
 * v3 说明：正常路径下 ProjectionEngine 自动维护 Read Model，本函数降级为：
 *   (a) completeTask / cancelTask 的安全兜底（ProjectionEngine 失败时）
 *   (b) rebuildTasksSheet_()（手动全量重建快捷函数）
 *
 * v3.1 新增：写完 Tasks 后，如果最终 status 是 DONE/CANCELLED，
 * 顺手清掉 ActiveTasks 里可能残留的那一行（见文件头注释"ActiveTasks 一致性补强"）。
 *
 * @param {string} taskId
 * @param {object} [knownTask]  已知的字段 partial object 或完整 task 对象
 */
function _materializeTaskRow_(taskId, knownTask) {
  var task = knownTask;
  if (!task) {
    task = deriveTaskState_()[taskId];
  }
  if (!task) return;

  upsertRowByKey_(TASKS_SHEET_NAME, 'task_id', taskId, task);

  // v3.1：ActiveTasks 一致性兜底
  var finalStatus = String(task.status || '').toUpperCase();
  if (finalStatus === 'DONE' || finalStatus === 'CANCELLED') {
    try {
      deleteRowByKey_('ActiveTasks', 'task_id', taskId); // 05_SheetUtils.gs；找不到行时静默返回false
    } catch (e) {
      Logger.log('[ProductivityModule] ActiveTasks 安全网清理失败（不影响主流程，Tasks已正确更新）: ' + e.message);
    }
  }
}

// ============ 内部工具 ============

function generateTaskId_() {
  var tz = Session.getScriptTimeZone();
  var today = Utilities.formatDate(new Date(), tz, 'yyyyMMdd');
  var uniqueSuffix = Utilities.getUuid().split('-')[0].toUpperCase();
  return 'TSK-' + today + '-' + uniqueSuffix;
}
