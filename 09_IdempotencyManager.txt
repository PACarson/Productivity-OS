/**
 * 09_IdempotencyManager.gs
 * Productivity OS v3.1 — 幂等性管理器（Task）
 *
 * 职责：确保每个创建操作恰好执行一次（Exactly-Once Semantics）。
 * 所有模块创建 Task 必须经过本层，禁止直接 EventBus.publish('TASK_CREATED')。
 *
 * ⚠️ 2026-07-03 拆分说明：这是 Personal AI Core 09_IdempotencyManager.gs 的
 * Task 部分，Inventory/Reminder 部分留在 Core 项目自己的副本里。
 *
 * v3.1 新增（2026-06-29，外部审计HIGH RISK 1，核实属实后采纳）：
 *
 * 原问题：去重校验（读 Read Model）和创建写入（写 Events + 更新 Read Model）是两个
 * 分离的步骤，之间没有锁保护。当 Telegram Webhook 并发重试（或用户快速连按）触发两个
 * 执行实例同时跑时，两个实例会同时读到"任务不存在"，然后都走创建路径，
 * 导致 Tasks Sheet 出现两行完全相同的任务（幂等机制失效）。
 *
 * 修复：在"读取-校验-写入"全程用 LockService.getScriptLock() 包住，
 * 强制串行化。第一个执行完毕、锁释放后，第二个进来时 DeduplicationEngine
 * 已经能在 Read Model 里找到了，直接走"已存在"路径返回。
 *
 * ⚠️ 锁说明：
 *  - getScriptLock() 是进程级（Script）锁，同一个 GAS 部署（本项目）下所有
 *    并发执行共用。2026-07-03 拆分后：这把锁只保护本项目自己的并发执行，
 *    不跨项目——Personal AI Core 项目的 Inventory 创建走它自己独立的锁，
 *    两边不互相阻塞，也不需要互相阻塞（不同 Sheet/identity 空间）。
 *  - waitLock(10000) = 最多等 10 秒，超时则抛 "Could not obtain lock" 并向上传播。
 *  - 锁统一在 finally 块释放，保证不论成功/失败/异常都不死锁。
 *  - 注意：GAS LockService 锁不可重入。_createTaskDirect_ 不尝试再获取
 *    ScriptLock，所以安全。
 *
 * 依赖：
 *   07_IdentityEngine, 08_DeduplicationEngine
 *   _createTaskDirect_ — 20_ProductivityModule.gs 里的全局函数（同项目）
 */

var IdempotencyManager = (function () {

  // ============ Task ============

  /**
   * 创建任务（幂等，并发安全）
   *
   * @param {string} title
   * @param {object} meta   { category, priority, due_date, recurring, context, budget, notes }
   * @param {string} chatId
   * @returns {{ task: object, created: boolean }}
   */
  function createTaskIfNotExists(title, meta, chatId) {
    meta = meta || {};

    var identity = IdentityEngine.generateTaskIdentity(
      chatId,
      title,
      meta.due_date   || '',
      meta.recurring  || '',
      meta.priority   || 'MEDIUM',
      meta.category   || 'GENERAL'
    );

    var lock = LockService.getScriptLock();
    var lockAcquired = false;
    try {
      lock.waitLock(10000); // 最多等10秒；超时会抛 "Could not obtain lock"
      lockAcquired = true;

      // 锁范围内：读-校验-写 原子化
      var existing = DeduplicationEngine.findExistingTask(identity);
      if (existing) {
        Logger.log('[IdempotencyManager] 任务已存在（并发安全），跳过创建: identity=' + identity.slice(0, 12) + '... task_id=' + existing.task_id);
        return { task: existing, created: false };
      }

      var task = _createTaskDirect_(title, meta, chatId, identity);
      return { task: task, created: true };

    } finally {
      if (lockAcquired) {
        lock.releaseLock();
      }
    }
  }

  // ============ 开发者测试 ============

  function testWebhookRetry() {
    Logger.log('=== IdempotencyManager.testWebhookRetry ===');
    var chatId = 'test_chat_' + new Date().getTime();
    var title  = '幂等测试任务-' + new Date().getTime();

    Logger.log('第一次创建...');
    var r1 = createTaskIfNotExists(title, { priority: 'LOW', category: 'GENERAL' }, chatId);
    Logger.log('结果1: created=' + r1.created + ' task_id=' + r1.task.task_id);

    Logger.log('第二次创建（模拟 Telegram Webhook 重试）...');
    var r2 = createTaskIfNotExists(title, { priority: 'LOW', category: 'GENERAL' }, chatId);
    Logger.log('结果2: created=' + r2.created + ' task_id=' + r2.task.task_id);

    var ok1 = (r1.task.task_id === r2.task.task_id);
    var ok2 = !r2.created;
    Logger.log('同一个 task_id? ' + ok1 + '  (expected: true)');
    Logger.log('第二次 created=false? ' + ok2 + '  (expected: true)');
    Logger.log('=== testWebhookRetry ' + (ok1 && ok2 ? 'PASS ✅' : 'FAIL ❌') + ' ===');
  }

  function testConcurrentExecution() {
    Logger.log('=== IdempotencyManager.testConcurrentExecution ===');
    Logger.log('v3.1 防线：ScriptLock（waitLock 10s）确保"读-校验-写"串行化。');
    Logger.log('三层总防线：');
    Logger.log('  防线1：ScriptLock             → 并发执行串行化（最强）');
    Logger.log('  防线2：EventBus._inExecIdentityCache_ → 同次执行内双写（补充）');
    Logger.log('  防线3：DeduplicationEngine.findExistingTask → 跨执行跨实例去重（兜底）');
    Logger.log('如需压力测试，手动在 GAS 编辑器同时触发两个执行，检查 Tasks Sheet 行数。');
    Logger.log('=== testConcurrentExecution DONE ===');
  }

  return {
    createTaskIfNotExists:   createTaskIfNotExists,
    testWebhookRetry:        testWebhookRetry,
    testConcurrentExecution: testConcurrentExecution
  };
})();
