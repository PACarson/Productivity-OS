// ============================================================
// 31_ReminderQueue.gs — Reminder Queue Manager
// All reminders pass through here before being sent
// Handles dedup, priority sorting, archiving
// ============================================================

const ReminderQueue = (() => {

  // Enqueue a reminder — dedup check built in
  // opts.bypassDedup — skip the dedup check. Used by SHIFT_CLOSED scans so a
  // completed shift always gets a fresh RIDER BRIEF, even if the same item
  // was already SENT within the DEDUP_HOURS window (fixes Bug-08: silent
  // /stop reply when /maintdue still shows items due).
  // opts.sentImmediately — write the row as already SENT (with SENT_AT set)
  // instead of PENDING. Used by callers that send via TelegramService
  // directly and only need ReminderQueue as the timestamp/history trail —
  // e.g. BookingEngine.checkEscalation() (P16) — so the item doesn't also
  // sit in getPending() waiting to be re-sent by a later flush().
  function enqueue(item, opts) {
    try {
      const tz  = Session.getScriptTimeZone();
      const ts  = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd HH:mm:ss');
      const C   = CONFIG.COL.REMINDER_QUEUE;
      const bypassDedup    = !!(opts && opts.bypassDedup);
      const sentImmediately = !!(opts && opts.sentImmediately);

      // Dedup check — same source+category+bike within DEDUP_HOURS
      if (!bypassDedup && _isDuplicate(item)) return;

      TruthEngine.append(CONFIG.SHEETS.REMINDER_QUEUE, [
        ts,
        item.source,
        item.type,
        item.category,
        item.bike || 'ALL',
        item.priority || 0,
        item.message  || '',
        sentImmediately ? CONFIG.REMINDER_STATUS.SENT : CONFIG.REMINDER_STATUS.PENDING,
        sentImmediately ? ts : '',
      ]);

    } catch (e) {
      console.error('ReminderQueue.enqueue: ' + e.message);
    }
  }

  // Get all PENDING items sorted by priority desc
  function getPending() {
    try {
      const rows = TruthEngine.getRows(CONFIG.SHEETS.REMINDER_QUEUE);
      const C    = CONFIG.COL.REMINDER_QUEUE;
      const pending = rows
        .map((row, i) => ({ row, rowIndex: i + 1 }))
        .filter(({ row }) => row[C.STATUS - 1] === CONFIG.REMINDER_STATUS.PENDING);

      pending.sort((a, b) =>
        Number(b.row[CONFIG.COL.REMINDER_QUEUE.PRIORITY - 1]) -
        Number(a.row[CONFIG.COL.REMINDER_QUEUE.PRIORITY - 1])
      );

      return pending;
    } catch (e) { return []; }
  }

  // Mark items as SENT
  function markSent(rowIndices) {
    try {
      const tz  = Session.getScriptTimeZone();
      const ts  = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd HH:mm:ss');
      const C   = CONFIG.COL.REMINDER_QUEUE;

      for (const rowIndex of rowIndices) {
        TruthEngine.updateRow(CONFIG.SHEETS.REMINDER_QUEUE, rowIndex, {
          [C.STATUS]:  CONFIG.REMINDER_STATUS.SENT,
          [C.SENT_AT]: ts,
        });
      }
    } catch (e) {
      console.error('ReminderQueue.markSent: ' + e.message);
    }
  }

  // Archive SENT items older than ARCHIVE_DAYS to Reminder_History
  function archive() {
    try {
      const rows     = TruthEngine.getRows(CONFIG.SHEETS.REMINDER_QUEUE);
      const C        = CONFIG.COL.REMINDER_QUEUE;
      const tz       = Session.getScriptTimeZone();
      const now      = new Date();
      const cutoff   = new Date(now.getTime() - CONFIG.REMINDER.ARCHIVE_DAYS * 86400000);
      const toDelete = [];

      for (let i = 0; i < rows.length; i++) {
        const row    = rows[i];
        const status = row[C.STATUS - 1];
        if (status !== CONFIG.REMINDER_STATUS.SENT) continue;

        const sentAt = row[C.SENT_AT - 1];
        if (!sentAt) continue;

        const sentDate = sentAt instanceof Date ? sentAt : new Date(sentAt);
        if (sentDate > cutoff) continue;

        // Move to Reminder_History
        const CH = CONFIG.COL.REMINDER_HISTORY;
        TruthEngine.append(CONFIG.SHEETS.REMINDER_HISTORY, [
          row[C.TIMESTAMP - 1],
          row[C.SOURCE    - 1],
          row[C.TYPE      - 1],
          row[C.CATEGORY  - 1],
          row[C.BIKE      - 1],
          row[C.PRIORITY  - 1],
          row[C.MESSAGE   - 1],
          Utilities.formatDate(now, tz, 'yyyy-MM-dd HH:mm:ss'),
        ]);

        toDelete.push(i + 1); // 1-based rowIndex
      }

      // Delete archived rows in reverse order
      if (toDelete.length > 0) {
        const sheet = TruthEngine.getSheet(CONFIG.SHEETS.REMINDER_QUEUE);
        for (let i = toDelete.length - 1; i >= 0; i--) {
          sheet.deleteRow(toDelete[i] + 1); // +1 for header
        }
      }

    } catch (e) {
      console.error('ReminderQueue.archive: ' + e.message);
    }
  }

  // ── Private — Dedup check ─────────────────────────────────

  // Minutes since the most recent PENDING/SENT item matching
  // source+category+bike, or null if none exists. Shared scan used by
  // both _isDuplicate() (fixed DEDUP_HOURS window) and any caller that
  // needs a dynamic interval instead — e.g. BookingEngine.checkEscalation()
  // (P16), which shrinks the required gap as a Critical slot's deadline
  // approaches instead of using one fixed window for everything.
  function getMinutesSinceLastMatch(item) {
    try {
      const rows = TruthEngine.getRows(CONFIG.SHEETS.REMINDER_QUEUE);
      const C    = CONFIG.COL.REMINDER_QUEUE;
      const now  = Date.now();
      let best   = null;

      for (const row of rows) {
        const status = row[C.STATUS - 1];
        // Check against both PENDING and SENT
        if (status !== CONFIG.REMINDER_STATUS.PENDING &&
            status !== CONFIG.REMINDER_STATUS.SENT) continue;

        if (row[C.SOURCE   - 1] !== item.source)   continue;
        if (row[C.CATEGORY - 1] !== item.category) continue;
        if (String(row[C.BIKE - 1]) !== String(item.bike || 'ALL')) continue;

        const ts  = row[C.TIMESTAMP - 1];
        const tsD = ts instanceof Date ? ts : new Date(ts);
        if (isNaN(tsD.getTime())) continue;

        const minsAgo = (now - tsD.getTime()) / 60000;
        if (best === null || minsAgo < best) best = minsAgo;
      }
      return best;
    } catch (e) {
      return null;
    }
  }

  function _isDuplicate(item) {
    const minsAgo = getMinutesSinceLastMatch(item);
    return minsAgo !== null && minsAgo < CONFIG.REMINDER.DEDUP_HOURS * 60;
  }

  // Clear all PENDING items for a specific bike+source
  // Called before scan() re-enqueues fresh maintenance data
  // Prevents duplicate notifications when scan runs multiple times per shift
  function clearPendingByBike(bike, source) {
    try {
      const rows  = TruthEngine.getRows(CONFIG.SHEETS.REMINDER_QUEUE);
      const C     = CONFIG.COL.REMINDER_QUEUE;
      const sheet = TruthEngine.getSheet(CONFIG.SHEETS.REMINDER_QUEUE);
      const toDelete = [];

      for (let i = 0; i < rows.length; i++) {
        if (rows[i][C.STATUS - 1] !== CONFIG.REMINDER_STATUS.PENDING)             continue;
        if (source && rows[i][C.SOURCE - 1] !== source)                            continue;
        if (bike   && String(rows[i][C.BIKE - 1]).toUpperCase()
                   !== String(bike).toUpperCase())                                  continue;
        toDelete.push(i + 1); // 1-based data rowIndex
      }

      // Delete in reverse order to preserve row indices
      for (let i = toDelete.length - 1; i >= 0; i--) {
        sheet.deleteRow(toDelete[i] + 1); // +1 for header row
      }
    } catch (e) {
      console.error('ReminderQueue.clearPendingByBike: ' + e.message);
    }
  }

  return { enqueue, getPending, markSent, archive, clearPendingByBike, getMinutesSinceLastMatch };
})();
