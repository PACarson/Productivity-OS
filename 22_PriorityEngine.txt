/**
 * 22_PriorityEngine.gs
 * Productivity OS v4.0 — Priority Engine（V4 新增）
 *
 * 职责：给一批已经在内存里的 task 对象算 Priority Score / Urgency Score，
 * 并给出优先级建议。纯函数——不读 Sheet，不读 Events，不写任何东西。
 * 输入永远是 12_TaskQueryEngine.gs 已经批量读出来的 task 数组。
 *
 * 架构铁律（00_Project_Constitution.gs P5）：
 *   Priority Engine 永远只生成建议，不会自动改任何任务的 priority 字段。
 *   要采纳建议，必须走 20_TaskEngine.updateTask() 由用户/上层逻辑主动调用。
 *
 * 信号来源（v4.0 范围，见 00_Project_State.gs"下一步"P2第3条关于未纳入
 * 信号的说明）：
 *   - Manual Priority（LOW/MEDIUM/HIGH/CRITICAL，用户自己设的基础权重）
 *   - Due Date（越接近现在，Urgency 越高）
 *   - Overdue Days（已经逾期的天数，权重最高）
 *   - Reminder Count（提醒过越多次，说明一直没处理，适度加权）
 *   - Recurring（recurring 任务通常是常规事务，适度降权，避免长期占据
 *     "高优先级"位置挤占真正紧急的一次性任务）
 */

var PriorityEngine = (function () {

  var MANUAL_WEIGHT = { 'CRITICAL': 100, 'HIGH': 70, 'MEDIUM': 40, 'LOW': 10 };

  /**
   * Urgency Score：只跟"时间"相关的信号，跟手动 priority 无关。
   * 逾期越久分越高；未逾期则越接近到期分越高；没有 due_date 则最低。
   *
   * @param {object} task
   * @param {Date} [now]  可选，测试用，默认 new Date()
   * @returns {number}  0-100
   */
  function computeUrgencyScore(task, now) {
    now = now || new Date();
    if (!task.due_date) return 5; // 没有到期日，给个很低的基础分，不是0（0容易被当成"绝对不急"）

    var due = parseDueDate_(String(task.due_date)); // 05_SheetUtils.gs
    if (!due || isNaN(due.getTime())) return 5;

    var diffMs = due.getTime() - now.getTime();
    var diffDays = diffMs / (24 * 60 * 60 * 1000);

    if (diffDays < 0) {
      // 已逾期：逾期越久越紧急，但设上限避免"逾期100天"和"逾期10000天"没有区分度
      var overdueDays = Math.min(-diffDays, 30);
      return round1_(70 + overdueDays); // 05_SheetUtils.gs，70~100
    }

    if (diffDays <= 1)  return 65;
    if (diffDays <= 3)  return 50;
    if (diffDays <= 7)  return 35;
    if (diffDays <= 30) return 20;
    return 10;
  }

  /**
   * Priority Score：手动 priority 权重 + Urgency Score 的加权和，
   * recurring 任务打个小折扣（见文件头信号说明）。
   *
   * @param {object} task
   * @param {Date} [now]
   * @returns {number}  0-100
   */
  function computePriorityScore(task, now) {
    var manual  = MANUAL_WEIGHT[String(task.priority || 'MEDIUM').toUpperCase()] || MANUAL_WEIGHT.MEDIUM;
    var urgency = computeUrgencyScore(task, now);

    var score = manual * 0.5 + urgency * 0.5;
    if (task.recurring) score *= 0.9;

    return round1_(Math.min(score, 100)); // 05_SheetUtils.gs
  }

  /**
   * 建议一个 priority 标签（永不自动写回，只是建议）。
   * @param {object} task
   * @returns {string}  'LOW'|'MEDIUM'|'HIGH'|'CRITICAL'
   */
  function suggestPriority(task) {
    var score = computePriorityScore(task);
    if (score >= 80) return 'CRITICAL';
    if (score >= 55) return 'HIGH';
    if (score >= 25) return 'MEDIUM';
    return 'LOW';
  }

  /**
   * 给一批 task 排序（Priority Score 降序），供 TaskQueryEngine.getPriorityTasks 用。
   * @param {object[]} tasks
   * @returns {object[]}  新数组，每个元素附加 _priority_score 字段（供展示用，
   *                       不落盘）
   */
  function rankByPriority(tasks) {
    var now = new Date();
    return tasks
      .map(function (t) {
        var copy = shallowCopy_(t); // 05_SheetUtils.gs
        copy._priority_score = computePriorityScore(t, now);
        return copy;
      })
      .sort(function (a, b) { return b._priority_score - a._priority_score; });
  }

  return {
    computeUrgencyScore:  computeUrgencyScore,
    computePriorityScore: computePriorityScore,
    suggestPriority:      suggestPriority,
    rankByPriority:       rankByPriority
  };
})();
